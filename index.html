Q2. Problem: You are given a string (STR) of length N, you have to print the count of all alphabet.(using maps)
Input: abcadeecfb
Output:
        a=2
        b=2
        c=2
        d=1
        e=2
        f=1

Answer:

            let a = 'abcadeecfb';
            function count(str) {
                let alphabet = new Map();
                for (let char of str) {
                    if (alphabet.has(char)) {
                        alphabet.set(char, alphabet.get(char) + 1);
                    }
                    else {
                        alphabet.set(char, 1);
                    }
                }
                console.log(alphabet);
                for (let [key, value] of alphabet) {
                    console.log(key + "=" + value);
                }
            }
            count(a);

Output: 
                a=2
                b=2
                c=2
                d=1
                e=2
                f=1
